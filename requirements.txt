# macOS Tools - Requirements
# Owned and maintained by Valk, Admin: PDXF

# Core libraries for utility tools
rich  # Beautiful and flexible Python library for command-line applications

# UI libraries (in case of future graphical tools)
tkinter  # Standard Python GUI library
customtkinter  # Modern and customizable Tkinter widgets

# HTTP and networking libraries
requests  # For making HTTP requests
urllib3  # For handling HTTP requests and connections

# File and directory handling
pathlib  # Standard library for manipulating filesystem paths (Python 3.4+)
os  # For interacting with the operating system

# Data manipulation
pandas  # Data analysis and manipulation library
numpy  # Core scientific computing library

# Text manipulation
pyperclip  # Clipboard text copy/paste utility
python-docx  # For reading/writing Word documents
python-pptx  # For working with PowerPoint files

# Miscellaneous utilities
pyyaml  # For parsing and writing YAML files
paramiko  # SSH2 protocol library for Python
psutil  # Cross-platform library for system and process utilities

# Additional utilities (for future expansion)
beautifulsoup4  # For web scraping tasks
lxml  # XML and HTML parsing
pillow  # Image processing library (for working with images)
flask  # Web framework for Python (for building web apps)
sqlite3  # SQLite database interface
pytest  # Testing framework for Python
